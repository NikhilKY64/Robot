// Arduino Human Following Robot
// Works with 3-pin IR sensors + Ultrasonic sensor
// Using AFMotor + NewPing + Servo libraries
// Maintains ~10 cm following distance smoothly
// Turning speed adjustable by single variable

#include <NewPing.h>
#include <Servo.h>
#include <AFMotor.h>

// ---------------- Pin Definitions ----------------
#define RIGHT A2
#define LEFT  A3
#define TRIGGER_PIN A1
#define ECHO_PIN    A0
#define MAX_DISTANCE 200

// ---------------- Adjustable Settings ----------------
#define TURN_SPEED 200   // ðŸ”§ Adjust turning speed here (150â€“255 recommended)
#define FWD_SPEED_MIN 100
#define FWD_SPEED_MAX 220
#define BACKWARD_SPEED 120

// ---------------- Globals ----------------
unsigned int distance = 0;    
unsigned int Right_Value = 0; 
unsigned int Left_Value  = 0; 

NewPing sonar(TRIGGER_PIN, ECHO_PIN, MAX_DISTANCE);

// Motors (Adafruit Shield)
AF_DCMotor Motor1(1, MOTOR12_1KHZ);
AF_DCMotor Motor2(2, MOTOR12_1KHZ);
AF_DCMotor Motor3(3, MOTOR34_1KHZ);
AF_DCMotor Motor4(4, MOTOR34_1KHZ);

// Servo
Servo myservo; 
int pos = 0;    

// ---------------- Setup ----------------
void setup() {
  Serial.begin(9600);
  myservo.attach(10);
  myservo.write(90);
  delay(500);

  pinMode(RIGHT, INPUT);
  pinMode(LEFT, INPUT);
}

// ---------------- Loop ----------------
void loop() {
  delay(50);                         
  distance = sonar.ping_cm();
  Right_Value = digitalRead(RIGHT);
  Left_Value  = digitalRead(LEFT);

  // --------- Movement Logic ---------
  if ((Right_Value == 0) && (Left_Value == 1)) {
    turnLeft(TURN_SPEED);   // Use fixed adjustable turn speed
  }
  else if ((Right_Value == 1) && (Left_Value == 0)) {
    turnRight(TURN_SPEED);  // Use fixed adjustable turn speed
  }
  else {
    if ((distance > 0) && (distance < 10)) {
      moveBackward(BACKWARD_SPEED);
    }
    else if ((distance >= 10) && (distance < 50)) {
      int speed = map(distance, 10, 50, FWD_SPEED_MIN, FWD_SPEED_MAX);
      moveForward(speed);
    }
    else {
      stopMotors();
    }
  }
}

// ---------------- Helper Functions ----------------
void moveForward(int speed) {
  Motor1.setSpeed(speed);
  Motor1.run(FORWARD);
  Motor2.setSpeed(speed);
  Motor2.run(FORWARD);
  Motor3.setSpeed(speed);
  Motor3.run(FORWARD);
  Motor4.setSpeed(speed);
  Motor4.run(FORWARD);
  Serial.print("Forward, speed="); Serial.println(speed);
}

void moveBackward(int speed) {
  Motor1.setSpeed(speed);
  Motor1.run(BACKWARD);
  Motor2.setSpeed(speed);
  Motor2.run(BACKWARD);
  Motor3.setSpeed(speed);
  Motor3.run(BACKWARD);
  Motor4.setSpeed(speed);
  Motor4.run(BACKWARD);
  Serial.print("Backward, speed="); Serial.println(speed);
}

void turnLeft(int speed) {
  Motor1.setSpeed(speed);
  Motor1.run(FORWARD);
  Motor2.setSpeed(speed);
  Motor2.run(FORWARD);
  Motor3.setSpeed(speed);
  Motor3.run(BACKWARD);
  Motor4.setSpeed(speed);
  Motor4.run(BACKWARD);
  Serial.print("Turning Left, speed="); Serial.println(speed);
}

void turnRight(int speed) {
  Motor1.setSpeed(speed);
  Motor1.run(BACKWARD);
  Motor2.setSpeed(speed);
  Motor2.run(BACKWARD);
  Motor3.setSpeed(speed);
  Motor3.run(FORWARD);
  Motor4.setSpeed(speed);
  Motor4.run(FORWARD);
  Serial.print("Turning Right, speed="); Serial.println(speed);
}

void stopMotors() {
  Motor1.run(RELEASE);
  Motor2.run(RELEASE);
  Motor3.run(RELEASE);
  Motor4.run(RELEASE);
  Serial.println("Stopped");
}
